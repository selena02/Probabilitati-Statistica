\subsection{Camp de probabilitate.Operatii cu evenimente.Formule de calcul}
\indent experiment aleator $\to$ ${(\Omega,\textit{F})}$\\
\indent \indent $\mathbb{P}$ : $\textit{F}$ $\to$ [0,1]\\
\indent Presupunem ca avem un experiment aleator si un eveniment A de interes. Repetam experimentul (in conditii similare) de un numar mare de ori(N).\\
\indent\indent $N_{A}$ - numar de realizari ale lui A\\
\indent\indent $\frac{N_{A}}{N}$ - frecventa relativa de realizare a lui A\\
\indent \indent $N_{A}$ = \{0,1,2,..,N\}\\
\indent\indent ${\mathbb{P} (A) \simeq}$ ${ \lim_{N \rightarrow  \infty } \frac{N(A)}{B}  }$ \\
\indent\indent ${\frac{N(A)}{N} \epsilon [0,1]}$ \\
\indent\indent ${\mathbb{P} (A) \epsilon [0,1]}$\\
Daca A = ${\Omega \ \Longrightarrow N (\Omega) = N  \Longrightarrow \frac{N(A)}{N} = 1 \Longrightarrow   \mathbb{P} (A) =1} $  \\
\indent ${\mathbb{P} (A) \epsilon [0,1]}$\\
\indent ${\mathbb{P} (\Omega) = 1}$\\
Presupunem ca A,B ${\epsilon \textit{F} , A \cap B = \emptyset}$\\
\indent ${A \cup B \epsilon \textit{F}}$\\
\indent ${N(A \cup B) = N(A) + N(B)}$\\
\indent ${\mathbb{P} (A \cup B) = \mathbb{P} (A) + \mathbb{P} (B)}$ - finit aditivitate\\
\includegraphics{c2_p1.png}\\\\

$\textbf{Definitie}$ O functie ${\mathbb{P} :  \textit{F} \to [0,1]}$ se numeste masura de probabilitate pe ${(\Omega, \textit{F})}$ daca verifica urmaroarele proprietati:\\
\indent a) ${\mathbb{P} (\Omega) = 1}$ \\
\indent b) oricare ar fi ${(A_{n})_{n}  \subseteq  \textit{F}}$ disjuncte 2 cate 2\\
Experiment aleator $\to$ ${(\Omega, \textit{F}, \mathbb{P} (B))}$ camp de probabilitate\\

\indent ${\mathbb{P}(A|B)=\frac{1}{4}}$\\
\indent B- evenimentul prin care la prima aruncare am obtinut H.\\
\indent ${\mathbb{P}(A|B)}$ - probabilitatea realizarii lui A stiind ca B s-a realizate ${||}$ probabilitatea conditionata a lui A la B\\\\
${\textbf{Din perspectiva frecventionala:}}$ Avem un experiment pe care il repetam de un numar N de ori. Ne intereseaza evenimentele A si B. \\\\
\indent\indent ${\frac{N(A \cap B)}{N(B)}=\frac{\frac{N(A \cap B)}{N}}{\frac{N(B)}{N}}  \simeq  \frac{\mathbb{P}(A \cap B)}{\mathbb{P}(B)}}$\\\\

${\textbf{Definitie.}}$ Fie ${(\Omega,\textit{F},\mathbb{P})}$ un camp de probabilitate , A,B ${\epsilon \textit{F}}$ cu ${\mathbb{P}(B) > 0}$ atunci definim probabilitatea conditionata a lui A la evenimentul B si notam ${\mathbb{P}(A|B)}$ prin ${\mathbb{P}(A|B)=\frac{\mathbb{P}(A \cap B)}{\mathbb{P}(B)}}$\\
\indent ${\mathbb{P}(A)}$ -prior sau probabilitate a priori\\
\indent ${\mathbb{P}(A|B)}$ -posterior sau probabilitate a posteriori\\

\indent ${\mathbb{P}(B|A)=\frac{\mathbb{P}(A \cap B)}{\mathbb{P}(A)}=\frac{\frac{13*12}{52*51}}{\frac{13}{52}}=\frac{12}{51}}$\\\\
\indent ${\mathbb{P}(C|A)=\frac{\mathbb{P}(A \cap C)}{\mathbb{P}(A)}=\frac{\frac{13*25}{52*51}}{\frac{13}{52}}=\frac{25}{51}}$\\\\
\indent ${\mathbb{P}(A|B)=\frac{\mathbb{P}(A \cap B)}{\mathbb{P}(B)}=\frac{\frac{13*12}{52*51}}{\frac{1}{4}}=\frac{12}{51}=\mathbb{P}(B|A)}$\\\\
\indent ${\mathbb{P}(A|C)=\frac{\mathbb{P}(A \cap C)}{\mathbb{P}(C)}=\frac{\frac{13*25}{52*51}}{\frac{25}{52}}=\frac{25}{102}\neq \mathbb{P}(C|A)}$\\\\

Doua evenimente sunt independente daca realizarea uneia nu aduce informatii suplimentare despre realizarea celeilalte.
${(\Omega,\textit{F},\mathbb{P}),\:A,B\epsilon\textit{F}}$\\
${\Longrightarrow \mathbb{P}(A|B)=\mathbb{P}(B|A)\Longleftrightarrow \frac{\mathbb{P}(A\cap B)}{\mathbb{P}(B)}=\mathbb{P}(A)\Longleftrightarrow \: \mathbb{P}(A \cap B )=\mathbb{P}(A)*\mathbb{P}(A)}$\\\\

${\textbf{Definitie.}}$ Fie ${(\Omega,\textit{F},\mathbb{P})}$ un camp de probabilitate ${A,B \epsilon \textit{F}}$. Spunema ca A si B sunt independente si notam ${A \indep B}$ daca ${\mathbb{P}(A\cap B) =\mathbb{P}(A)*\mathbb{P}(B)}$.\\

${\textbf{P.}\: X\indep Y \Leftrightarrow \mathbb{P}(X\in A,Y\in B\mathbb{P}(X\in A)*\mathbb{P}(Y \in B)\:,\forall A,B\subseteq\mathbb{R}(interval)}$\\
${\textbf{P.}}$ Daca X si Y variabile aleatoare astfel incat ${X\indep Y}$ si g si h doua functii, atunci ${g(x)\indep h(y)}$\\
${\textbf{Definitie.}\: X_1,X_2,..,X_n}$ sunt independente daca:\\
${\mathbb{P}(X_1\leq x_1,X_2\leq x_2,..,X_n\leq x_n)=\mathbb{P}(X_1\leq x_1)*\mathbb{P}(X_2\leq x_2)*..*\mathbb{P}(X_n\leq x_n),\: \forall x_1,x_2,..,x_n\in \mathbb{R}}$\\

$\textbf{Definitie.}$
Fie X si Y doua v.a se numesc covariante dintre X si Y\\
$Cov(X,Y) =\mathbb{E}{[(X-\mathbb{E}{[X]})(Y-\mathbb{E}{[Y]})]}$\\
Convarianta este $\mathbb{E}{[XY]}-\mathbb{E}{[X]}\cdot\mathbb{E}{[Y]}$\\
$\textbf {Definitie}$\\
Spunem ca X si Y sunt necorelate daca convarianta lor este 0
Daca $X\perp Y \Rightarrow $ X si Y sunt necorelate\\
$\textbf{Proprietati.}$//
a) Cov(X,X) = Var(X)\\
b) Cov(X,a) =0 , a const\\
c) Cov(a+bx, y)=b Cov(X,Y)\\
d) Cov(X,Y) = Cov(Y,X)\\
e) Var(X+Y)=Var(X)+Var(Y)+2Cov(X,Y)\\
$Var(x_1 \cdots x_n)=\sum\limits _{i=1} Var(xi)+2\sum \limits_{i<j}Cov(xi, xj) $\\
f) Cov(x+y,z)=Cov(x,z)+Cov(y,z)\\
$\textbf {Definitie}$\\
Fie X si Y doua v.a si definim coeficientul de corelatie dintre X si Y\\
$\rho(X,Y) = \frac{Cov(X,Y)}{Var(X)\sqrt{Var(Y)}} $ \\\\

$X\sim \textit{U}([a,b])$ indep de $Y\sim \textit{U}([a,c])$\\
$f_x (x)=\frac{1}{b-a}* \mathds{1}_{[a,c]} (y)$\\
$f_{(x,y)} (x,y)=f_x (x) * f_y (y)=\frac{1}{a(b-a)}* \mathds{1}_{[a,c]} (x,y)$\\

\subsection{Teorema limita centrala(TLC)}
\noindent Din LNM : $\overline{X_n}\xrightarrow{\mathbb{P}}\mathbb{E}[X_n]$\\
Fie $x_1,..,x_n$ i.i.d. $\mathbb{E}[x_1]=\mu,\: Var(x_1)=\sigma^2$\\
${\frac{X-\mathbb{E}[X]}{\sqrt{Var(X)}}=1}$\\
Z = $\frac{X-\mathbb{E}[X]}{\sqrt{Var(X)}}=1 $ - Z s.n. variabila normalizata\\
$Z_n=\frac{x_1+..x+n-\mathbb{E}[x_1+..+x_n]}{\sqrt{Var(x_1+..+x_n)}}=\frac{x_1+..+x_n- n*\mu}{\sqrt{n\sigma^2}}$ - variabila de scor\\
$Z_n =\sqrt{n}(\frac{\overline{X_n}-\mu}{\sigma})$\\\\

$\textbf{TLC}$\\
Fie $(X_n)_n$ sir de v.a. i.i.d. $\mathbb{E}[X_1]=\mu<\infty,\: Var(X_1)=\sigma^2<\infty$.\\
Atunci $\lim_n \mathbb{P}(Z_n\leq x)=\phi(x) \: \forall x$\\
\indent unde $Z_n=\sqrt{x}*\frac{\overline{X_n}-\mu}{\sigma}\Longrightarrow \phi(x) = \int^{\infty}_{-\infty} \frac{1}{\sqrt{2\pi}}* e^{-t^2/2} dt$(functia de repartitie a N(0,1)\\
\\
$\textbf{Observatie}. \: x_1,x_1,..,x_n$ i.i.d. , $\mathbb{E}[x_i]=\mu,\: Var(x_i)=\sigma^2,\: S_n=x_1+x_2+..+x_n$\\
$\mathbb{P}(S_n\leq c)=\mathbb{P}(\frac{S_n-\mathbb{E}[S_n]}{Var(S_n)}\leq \frac{c-\mathbb{E}[S_n]}{\sqrt{Var(S_n)}})=\mathbb{P}(Z_n\leq \frac{c* n\mu}{\sqrt{n\sigma^2}})\stackrel{TLC}{\simeq} \phi(\frac{x-n\mu}{\sigma\sqrt{n}})$\\
